程序

#in this case it is "aeiou" , all the Vowel letters

datafile=open("dictionary.txt","r")

def cleanWord(word):
	"""Return word in lower case without whitespace"""
	return word.strip().lower()
	
def getVowelsInword(word):
	vowelsStr="aeiou"
	vowelsInword=""
	for char in word:
		if char in vowelsStr:
			vowelsInword+=char
	return vowelsInword

#main program
