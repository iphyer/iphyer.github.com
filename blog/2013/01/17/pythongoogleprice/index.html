def creatlist(Gfile):
	datalist=[]
	sumall=0
	
	for line in Gfile:
		if line[0:4]=='Date':
			continue
		datanew=line.strip().split(',')
		vol=int(datanew[5])
		adj=float(datanew[6])
		yearmonth=datanew[0][0:7]
		monthsum=vol+adj
		pricetuple=(yearmonth,monthsum)
		datalist.append(pricetuple)
	return datalist
